"""Module that is used by the cart to send requests to the
archive interface
"""

from __future__ import absolute_import
from StringIO import StringIO
import pycurl
from cart.cart_env_globals import  ARCHIVE_INTERFACE_URL



[docs]class ArchiveRequests(object):
    """class that supports all the requests to the archive
    interface
    """

    def __init__(self):
        self._url = ARCHIVE_INTERFACE_URL


[docs]    def pull_file(self, archive_filename, cart_filepath):
        """Performs a curl that will attempt to write
        the contents of a file from the archive interface
        to the specified cart filepath
        """
        mycurl = pycurl.Curl()
        mycurl.setopt(pycurl.URL, str(self._url + archive_filename))
        with open(cart_filepath, 'w+') as myfile:
            mycurl.setopt(pycurl.WRITEFUNCTION, myfile.write)
            mycurl.perform()
        mycurl.close()



[docs]    def stage_file(self, file_name):
        """Sends a post to the archive interface telling it to stage the file
        """

        mycurl = pycurl.Curl()
        mycurl.setopt(pycurl.URL, str(self._url + file_name))
        mycurl.setopt(pycurl.POST, True)
        mycurl.perform()
        mycurl.close()




[docs]    def status_file(self, file_name):
        """Gets a status from the  archive interface via Head and
        returns response """

        storage = StringIO()
        mycurl = pycurl.Curl()
        mycurl.setopt(pycurl.CUSTOMREQUEST, "HEAD")
        mycurl.setopt(pycurl.URL, str(self._url + file_name))
        mycurl.setopt(pycurl.NOBODY, False)
        mycurl.setopt(pycurl.WRITEFUNCTION, storage.write)
        mycurl.perform()
        mycurl.close()
        content = storage.getvalue()
        return content
