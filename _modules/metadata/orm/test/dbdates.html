SAMPLE_ZERO_ISO_HASH = {
    'created': '1969-12-31T16:00:00',
    'updated': '1969-12-31T16:00:00',
    'deleted': '1969-12-31T16:00:00'
}

    'created': 0,

    'created': datetime.now().replace(microsecond=0).isoformat(),

    'deleted': '1969-12-31T16:00:00'
}
SAMPLE_BAD_HASH = {
    'created': 'blarg',
    'updated': int(mktime(datetime.now().timetuple())),
    'deleted': None
}
